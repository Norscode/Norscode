# Norscode

Norscode er eit norsk språk- og verktøysett med native-first CLI, selfhost-løype og ei aktiv flate utan Python eller C.

![Norscode overview](docs/assets/norscode-overview.svg)

![Norscode logo](frontend/assets/icons/norscode-logo-dark.svg)

## Kort fortalt

- Norsk syntaks for funksjonar, kontrollflyt og uttrykk
- Statisk typing for heiltal, tekst, bool, lister og ordbøker
- Modul- og pakke-system
- Standardbiblioteket `std`
- `std.ai` for lokal AI-kontrakt: chat, embeddings, moderering, verktøy og agent-plan
- Feilhandtering med `kast`, `prøv` og `fang`
- Normal kjede: `.no` -> NCB JSON -> `selfhost/vm.no`

## std/fil.no

```
# fil.no — kompatibilitetslag for eldre `std.fil`-importar
#
# Repoet har i praksis filhjelparar fordelt på `std.filops` og `std.lagring`.
# Denne modulen gir ei liten fasade for testane som framleis bruker
# `bruk std.fil som fil`.

bruk std.filops som filops

funksjon finnes(sti: tekst) -> bool {
    prøv {
        builtin.fil_les(sti)
        returner sann
    } fang (e) {
        returner usann
    }
}

funksjon skriv_fil(sti: tekst, innhald: tekst) -> heltall {
    prøv {
        builtin.fil_skriv(sti, innhald)
        returner lengde(innhald)
    } fang (e) {
        returner 0
    }
}

funksjon les_fil(sti: tekst) -> tekst {
    prøv {
        returner builtin.fil_les(sti)
    } fang (e) {
        returner ""
    }
}

funksjon slett_fil(sti: tekst) -> bool {
    returner filops.slett_fil(sti)
}

funksjon status(sti: tekst) -> ordbok_tekst {
    la r: ordbok_tekst = {}
    hvis (sti != null og sti != "") {
        r["gyldig"] = "sann"
    } ellers {
        r["gyldig"] = "usann"
    }
    prøv {
        builtin.fil_les(sti)
        r["finnes"] = "sann"
    } fang (e) {
        r["finnes"] = "usann"
    }
    r["sti"] = sti
    returner r
}

```

## compiler/compiler.no

```
pakke compiler

bruk std.runtime
bruk std.compiler

struktur CompilerRuntimeState {
    allocator: Allocator
    bootstrap_ok: bool
    compiler_flags: heiltall
}

funksjon init_runtime() -> CompilerRuntimeState {
    skrivlinje("[compiler.no] initialiser runtime heap")

    la allocator = init_allocator()

    returner CompilerRuntimeState {
        allocator: allocator,
        bootstrap_ok: usann,
        compiler_flags: 0
    }
}

funksjon start_parser_runtime() {
    skrivlinje("[compiler.no] start parser runtime")
}

funksjon start_semantic_runtime() {
    skrivlinje("[compiler.no] start semantic runtime")
}

funksjon start_lowering_runtime() {
    skrivlinje("[compiler.no] start lowering runtime")
}

funksjon start_backend_runtime() {
    skrivlinje("[compiler.no] start backend runtime")
}

funksjon start_linker_runtime() {
    skrivlinje("[compiler.no] start linker runtime")
}

funksjon run_native_pipeline(
    state: CompilerRuntimeState
) -> bool {
    skrivlinje("[compiler.no] start native compiler pipeline")

    start_parser_runtime()
    start_semantic_runtime()
    start_lowering_runtime()
    start_backend_runtime()
    start_linker_runtime()

    la output = ny_runtime_string(
        state.allocator,
        "build/norscode-compiler"
    )

    returner bootstrap_compile(
        state.allocator,
        output
    )
}

funksjon compile_self(
    state: CompilerRuntimeState
) -> bool {
    skrivlinje("[compiler.no] compile compiler.no")

    returner run_native_pipeline(state)
}

funksjon rebuild_with_selfhost(
    state: CompilerRuntimeState
) -> bool {
    skrivlinje("[compiler.no] rebuild with selfhost compiler")

    hvis ikkje state.bootstrap_ok {
        returner usann
    }

    returner compile_self(state)
}

funksjon hoved() -> heiltall {
    skrivlinje("[compiler.no] start selfhost compiler")

    la runtime_state = init_runtime()

    runtime_state.bootstrap_ok = compile_self(
        runtime_state
    )

    hvis runtime_state.bootstrap_ok {
        skrivlinje("[compiler.no] bootstrap compile OK")

        la selfhost_ok = rebuild_with_selfhost(
            runtime_state
        )

        hvis selfhost_ok {
            skrivlinje("[compiler.no] selfhost rebuild OK")
            returner 0
        }
    }

    skrivlinje("[compiler.no] compiler failure")

    returner 1
}

```
